Creating the same networkd for the docker containers
docker network create pg-network

Start pg-database in docker #### this is the first container
winpty docker run -it \
  -e POSTGRES_USER="root" \
  -e POSTGRES_PASSWORD="root" \
  -e POSTGRES_DB="ny_taxi"\
  -v //c/Users/x-Black/Documents/Learning/DataEngineering/pg_data:/var/lib/postgresql/data \
  -p 5432:5432 \
 --network=pg-network \
 --name pg-database \
  postgres:13
  
Start the pgAdmin server   #### this is the second container
winpty docker run -it \
    -e PGADMIN_DEFAULT_EMAIL="admin@admin.com" \
    -e PGADMIN_DEFAULT_PASSWORD="root" \
    -p 8080:80 \
    --network=pg-network \
    --name pgadmin \
    dpage/pgadmin4
